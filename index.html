package com.susworld.chat;
import com.susworld.chat.chat.ChatManager;
import com.susworld.chat.command.AdCommand;
import com.susworld.chat.command.SUSChatCommand;
import com.susworld.chat.config.ConfigManager;
import com.susworld.chat.filter.AntiSpam;
import com.susworld.chat.filter.SwearFilter;
import com.susworld.chat.listener.ChatListener;
import org.bukkit.plugin.java.JavaPlugin;
import java.util.logging.Level;
/**
 * Главный класс плагина SUSChat для SUS World 1.21.1
 * Чат-система с глобальным/локальным чатом, объявлениями, фильтром мата
 */
public class SUSChatPlugin extends JavaPlugin {
    private static SUSChatPlugin instance;
    private ConfigManager configManager;
    private ChatManager chatManager;
    private SwearFilter swearFilter;
    private AntiSpam antiSpam;
    @Override
    public void onEnable() {
        instance = this;
        // Сохраняем дефолтный конфиг
        saveDefaultConfig();
        // Инициализация менеджеров
        this.configManager = new ConfigManager(this);
        this.swearFilter = new SwearFilter(this);
        this.antiSpam = new AntiSpam(this);
        this.chatManager = new ChatManager(this);
        // Регистрация слушателей
        getServer().getPluginManager().registerEvents(
            new ChatListener(this), this
        );
        // Регистрация команд
        getCommand("ad").setExecutor(new AdCommand(this));
        getCommand("suschat").setExecutor(new SUSChatCommand(this));
        // ASCII арт в консоль
        printStartupBanner();
        getLogger().info("SUSChat v" + getDescription().getVersion() + " успешно загружен!");
    }
    @Override
    public void onDisable() {
        getLogger().info("SUSChat выключен. До встречи!");
        instance = null;
    }
    /**
     * Перезагрузка всех компонентов плагина
     */
    public void reload() {
        reloadConfig();
        configManager.reload();
        swearFilter.reload();
        antiSpam.reload();
        chatManager.reload();
        getLogger().info("Конфигурация перезагружена!");
    }
    private void printStartupBanner() {
        getLogger().log(Level.INFO, "");
        getLogger().log(Level.INFO, "  ╔═══════════════════════════════╗");
        getLogger().log(Level.INFO, "  ║     SUS World Chat v2.0      ║");
        getLogger().log(Level.INFO, "  ║     Minecraft 1.21.1         ║");
        getLogger().log(Level.INFO, "  ╚═══════════════════════════════╝");
        getLogger().log(Level.INFO, "");
    }
    // ── Getters ──────────────────────────────────────────
    public static SUSChatPlugin getInstance() { return instance; }
    public ConfigManager getConfigManager() { return configManager; }
    public ChatManager getChatManager() { return chatManager; }
    public SwearFilter getSwearFilter() { return swearFilter; }
    public AntiSpam getAntiSpam() { return antiSpam; }
}

package com.susworld.chat.chat;
import com.susworld.chat.SUSChatPlugin;
import com.susworld.chat.util.ColorUtil;
import com.susworld.chat.util.SoundUtil;
import net.kyori.adventure.text.Component;
import net.kyori.adventure.text.minimessage.MiniMessage;
import net.kyori.adventure.title.Title;
import org.bukkit.Bukkit;
import org.bukkit.Sound;
import org.bukkit.entity.Player;
import java.time.Duration;
import java.util.*;
/**
 * Менеджер чата — обрабатывает отправку глобальных, локальных
 * сообщений и объявлений
 */
public class ChatManager {
    private final SUSChatPlugin plugin;
    private final MiniMessage miniMessage;
    private final ChatFormatter formatter;
    // Кулдаун объявлений (UUID -> timestamp)
    private final Map<UUID, Long> adCooldowns = new HashMap<>();
    public ChatManager(SUSChatPlugin plugin) {
        this.plugin = plugin;
        this.miniMessage = MiniMessage.miniMessage();
        this.formatter = new ChatFormatter(plugin);
    }
    /**
     * Отправляет глобальное сообщение всем игрокам на сервере
     */
    public void sendGlobal(Player sender, String message) {
        String formatted = formatter.format(ChatType.GLOBAL, sender, message);
        Component component = miniMessage.deserialize(formatted);
        // Обработка упоминаний
        for (Player online : Bukkit.getOnlinePlayers()) {
            online.sendMessage(component);
            checkMention(sender, online, message);
        }
        // Логгирование в консоль
        plugin.getLogger().info("[G] " + sender.getName() + ": " + message);
    }
    /**
     * Отправляет локальное сообщение игрокам в радиусе
     */
    public void sendLocal(Player sender, String message) {
        int radius = plugin.getConfig().getInt("chat.local-radius", 100);
        String formatted = formatter.format(ChatType.LOCAL, sender, message);
        Component component = miniMessage.deserialize(formatted);
        boolean anyoneHeard = false;
        for (Player online : Bukkit.getOnlinePlayers()) {
            if (online.getWorld().equals(sender.getWorld())) {
                double distance = online.getLocation().distance(sender.getLocation());
                if (distance <= radius) {
                    online.sendMessage(component);
                    checkMention(sender, online, message);
                    if (!online.equals(sender)) anyoneHeard = true;
                }
            }
        }
        // Если никто не услышал — уведомить отправителя
        if (!anyoneHeard) {
            String nobodyMsg = plugin.getConfig().getString("chat.nobody-nearby",
                "<gray>Рядом никого нет...");
            sender.sendMessage(miniMessage.deserialize(nobodyMsg));
        }
        plugin.getLogger().info("[L] " + sender.getName() + ": " + message);
    }
    /**
     * Отправляет объявление всему серверу
     * @return true если объявление отправлено, false если кулдаун
     */
    public boolean sendAd(Player sender, String message) {
        // Проверка кулдауна
        if (!sender.hasPermission("suschat.ad.bypass-cooldown")) {
            long remaining = getAdCooldownRemaining(sender);
            if (remaining > 0) {
                String cooldownMsg = plugin.getConfig()
                    .getString("messages.ad-cooldown", "<red>Подожди {time} сек.")
                    .replace("{time}", String.valueOf(remaining));
                sender.sendMessage(miniMessage.deserialize(
                    plugin.getConfig().getString("messages.prefix", "") + cooldownMsg));
                return false;
            }
        }
        // Проверка минимальной длины
        int minLen = plugin.getConfig().getInt("ad.min-length", 10);
        if (message.length() < minLen) {
            String tooShortMsg = plugin.getConfig()
                .getString("messages.ad-too-short", "<red>Слишком коротко!")
                .replace("{min}", String.valueOf(minLen));
            sender.sendMessage(miniMessage.deserialize(
                plugin.getConfig().getString("messages.prefix", "") + tooShortMsg));
            return false;
        }
        // Форматирование объявления
        String header = plugin.getConfig().getString("ad.format.header", "");
        String content = plugin.getConfig().getString("ad.format.content", "{message}")
            .replace("{player}", sender.getName())
            .replace("{message}", message);
        String footer = plugin.getConfig().getString("ad.format.footer", "");
        Component adComponent = miniMessage.deserialize(
            header + "<newline>" + content + "<newline>" + footer);
        // Отправка всем
        for (Player online : Bukkit.getOnlinePlayers()) {
            online.sendMessage(adComponent);
            // Звук
            if (plugin.getConfig().getBoolean("ad.sound.enabled", true)) {
                SoundUtil.playSound(online,
                    plugin.getConfig().getString("ad.sound.type", "ENTITY_EXPERIENCE_ORB_PICKUP"),
                    (float) plugin.getConfig().getDouble("ad.sound.volume", 1.0),
                    (float) plugin.getConfig().getDouble("ad.sound.pitch", 1.2));
            }
            // Title
            if (plugin.getConfig().getBoolean("ad.title.enabled", true)) {
                showAdTitle(online, sender.getName());
            }
        }
        // Установка кулдауна
        adCooldowns.put(sender.getUniqueId(), System.currentTimeMillis());
        plugin.getLogger().info("[AD] " + sender.getName() + ": " + message);
        return true;
    }
    /**
     * Показывает Title при объявлении
     */
    private void showAdTitle(Player player, String senderName) {
        String titleText = plugin.getConfig()
            .getString("ad.title.text", "<gold>📢 Объявление!");
        String subtitleText = plugin.getConfig()
            .getString("ad.title.subtitle", "<gray>от {player}")
            .replace("{player}", senderName);
        int fadeIn = plugin.getConfig().getInt("ad.title.fade-in", 10);
        int stay = plugin.getConfig().getInt("ad.title.stay", 40);
        int fadeOut = plugin.getConfig().getInt("ad.title.fade-out", 10);
        Title title = Title.title(
            miniMessage.deserialize(titleText),
            miniMessage.deserialize(subtitleText),
            Title.Times.times(
                Duration.ofMillis(fadeIn * 50L),
                Duration.ofMillis(stay * 50L),
                Duration.ofMillis(fadeOut * 50L)
            )
        );
        player.showTitle(title);
    }
    /**
     * Проверяет упоминание игрока в сообщении
     */
    private void checkMention(Player sender, Player target, String message) {
        if (sender.equals(target)) return;
        if (!plugin.getConfig().getBoolean("mentions.enabled", true)) return;
        String symbol = plugin.getConfig().getString("mentions.symbol", "@");
        if (message.toLowerCase().contains(
                symbol + target.getName().toLowerCase())) {
            if (plugin.getConfig().getBoolean("mentions.sound.enabled", true)) {
                SoundUtil.playSound(target,
                    plugin.getConfig().getString("mentions.sound.type",
                        "BLOCK_NOTE_BLOCK_PLING"),
                    (float) plugin.getConfig().getDouble("mentions.sound.volume", 1.0),
                    (float) plugin.getConfig().getDouble("mentions.sound.pitch", 1.5));
            }
        }
    }
    /**
     * Возвращает оставшийся кулдаун объявления в секундах
     */
    public long getAdCooldownRemaining(Player player) {
        if (!adCooldowns.containsKey(player.getUniqueId())) return 0;
        long lastUsed = adCooldowns.get(player.getUniqueId());
        long cooldownMs = plugin.getConfig().getInt("ad.cooldown", 300) * 1000L;
        long elapsed = System.currentTimeMillis() - lastUsed;
        if (elapsed >= cooldownMs) return 0;
        return (cooldownMs - elapsed) / 1000;
    }
    public void reload() {
        formatter.reload();
    }
    public ChatFormatter getFormatter() { return formatter; }
}

package com.susworld.chat.chat;
/**
 * Типы чата в системе SUSChat
 */
public enum ChatType {
    GLOBAL("global", "Глобальный"),
    LOCAL("local", "Локальный"),
    AD("ad", "Объявление");
    private final String key;
    private final String displayName;
    ChatType(String key, String displayName) {
        this.key = key;
        this.displayName = displayName;
    }
    public String getKey() { return key; }
    public String getDisplayName() { return displayName; }

     * Форматирует сообщение для указанного типа чата

            case GLOBAL -> globalFormat;
            case LOCAL -> localFormat;
            default -> "{message}";
        };

        message = processMentions(message);
        // Замена плейсхолдеров
        return template
            .replace("{player}", player.getName())
            .replace("{message}", message)
            .replace("{world}", player.getWorld().getName())
            .replace("{prefix}", getPrefix(player))
            .replace("{suffix}", getSuffix(player));

        }
        return message;
    }

import com.susworld.chat.SUSChatPlugin;

/**
 * Слушатель событий чата
 * Перехватывает все сообщения и маршрутизирует их
 */
public class ChatListener implements Listener {
    private final SUSChatPlugin plugin;
    private final MiniMessage miniMessage;
    public ChatListener(SUSChatPlugin plugin) {
        this.plugin = plugin;
        this.miniMessage = MiniMessage.miniMessage();
    }
    @EventHandler(priority = EventPriority.HIGHEST, ignoreCancelled = true)
    public void onChat(AsyncChatEvent event) {
        // Отменяем стандартную обработку чата
        event.setCancelled(true);
        Player player = event.getPlayer();
        String rawMessage = PlainTextComponentSerializer.plainText()
            .serialize(event.message());
        // ── Анти-спам ─────────────────────────────────────
        AntiSpam antiSpam = plugin.getAntiSpam();
        if (!player.hasPermission("suschat.bypass.spam")) {
            AntiSpam.SpamResult spamResult = antiSpam.check(player, rawMessage);
            if (spamResult != AntiSpam.SpamResult.OK) {
                String msg = switch (spamResult) {
                    case COOLDOWN -> plugin.getConfig()
                        .getString("antispam.messages.cooldown", "<red>Подожди!");
                    case CAPS -> plugin.getConfig()
                        .getString("antispam.messages.caps", "<red>Много капса!");
                    case REPEAT -> plugin.getConfig()
                        .getString("antispam.messages.repeat", "<red>Повтор!");
                    default -> "";
                };
                player.sendMessage(miniMessage.deserialize(
                    plugin.getConfig().getString("messages.prefix", "") + msg));
                return;
            }
        }
        // ── Фильтр мата ──────────────────────────────────
        String message = rawMessage;
        SwearFilter filter = plugin.getSwearFilter();
        if (filter.isEnabled() && !player.hasPermission("suschat.bypass.filter")) {
            SwearFilter.FilterResult result = filter.filter(message);
            message = result.filteredMessage();
            if (result.wasFiltered()) {
                String warnMsg = plugin.getConfig().getString(
                    "filter.warn-message", "<red>Мат заблокирован.");
                player.sendMessage(miniMessage.deserialize(warnMsg));
            }
        }
        // ── Убираем MiniMessage теги от обычных игроков ──
        if (!player.hasPermission("suschat.chat.color")) {
            message = stripMiniMessageTags(message);
        }
        // ── Определяем тип чата ──────────────────────────
        String globalPrefix = plugin.getConfig()
            .getString("chat.global-prefix", "!");
        if (message.startsWith(globalPrefix)) {
            // Глобальный чат
            if (!player.hasPermission("suschat.chat.global")) {
                player.sendMessage(miniMessage.deserialize(
                    plugin.getConfig().getString("messages.prefix", "") +
                    plugin.getConfig().getString("messages.no-permission",
                        "<red>Нет прав!")));
                return;
            }
            String globalMessage = message.substring(globalPrefix.length()).trim();
            if (!globalMessage.isEmpty()) {
                plugin.getChatManager().sendGlobal(player, globalMessage);
            }
        } else {
            // Локальный чат (по умолчанию)
            plugin.getChatManager().sendLocal(player, message);
        }
    }
    /**
     * Убирает MiniMessage теги из сообщения
     */
    private String stripMiniMessageTags(String message) {
        return message.replaceAll("<[^>]+>", "");
    }
}

package com.susworld.chat.filter;
import com.susworld.chat.SUSChatPlugin;
import java.util.*;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
/**
 * Умный фильтр русского мата
 *
 * Возможности:
 * - Находит мат даже через символы-разделители (пр0б3лы, т0чки)
 * - Замена символов-аналогов (0=о, 3=з, @=а, etc)
 * - Заменяет на # символы по длине слова
 * - Возвращает FilterResult с информацией
 */
public class SwearFilter {
    private final SUSChatPlugin plugin;
    private List<Pattern> compiledPatterns;
    private String replacementChar;
    private boolean enabled;
    // Маппинг символов-замен (leet speak для русского)
    private static final Map<Character, Character> CHAR_MAP = Map.ofEntries(
        Map.entry('0', 'о'), Map.entry('o', 'о'),
        Map.entry('@', 'а'), Map.entry('a', 'а'),
        Map.entry('e', 'е'), Map.entry('3', 'з'),
        Map.entry('u', 'и'), Map.entry('y', 'у'),
        Map.entry('x', 'х'), Map.entry('k', 'к'),
        Map.entry('p', 'р'), Map.entry('c', 'с'),
        Map.entry('6', 'б'), Map.entry('b', 'б'),
        Map.entry('d', 'д'), Map.entry('n', 'н'),
        Map.entry('m', 'м'), Map.entry('t', 'т'),
        Map.entry('h', 'н'), Map.entry('l', 'л'),
        Map.entry('ё', 'е')
    );
    public SwearFilter(SUSChatPlugin plugin) {
        this.plugin = plugin;
        reload();
    }
    /**
     * Перезагружает список слов из конфига и компилирует паттерны
     */
    public void reload() {
        enabled = plugin.getConfig().getBoolean("filter.enabled", true);
        replacementChar = plugin.getConfig().getString("filter.replacement-char", "#");
        List<String> words = plugin.getConfig().getStringList("filter.words");
        compiledPatterns = new ArrayList<>();
        for (String word : words) {
            String regex = buildSmartRegex(word);
            try {
                compiledPatterns.add(Pattern.compile(regex,
                    Pattern.CASE_INSENSITIVE | Pattern.UNICODE_CASE));
            } catch (Exception e) {
                plugin.getLogger().warning(
                    "Ошибка компиляции паттерна для: " + word);
            }
        }
        plugin.getLogger().info("Загружено " + compiledPatterns.size()
            + " паттернов фильтра мата");
    }
    /**
     * Строит умный regex из слова, разрешающий разделители между буквами
     * Пример: "хуй" -> "х[\\s._\\-*]*у[\\s._\\-*]*й"
     */
    private String buildSmartRegex(String word) {
        StringBuilder sb = new StringBuilder();
        String separators = "[\\s._\\-*!@#$%^&()=+|/\\\\,;:'\"]*";
        for (int i = 0; i < word.length(); i++) {
            char c = Character.toLowerCase(word.charAt(i));
            if (c == ' ') {
                sb.append(separators);
                continue;
            }
            // Собираем все возможные символы для этой буквы
            Set<Character> alternatives = new LinkedHashSet<>();
            alternatives.add(c);
            // Обратный поиск по маппингу
            for (Map.Entry<Character, Character> entry : CHAR_MAP.entrySet()) {
                if (entry.getValue() == c) {
                    alternatives.add(entry.getKey());
                }
            }
            // Прямой поиск
            if (CHAR_MAP.containsKey(c)) {
                alternatives.add(CHAR_MAP.get(c));
            }
            // ё/е взаимозаменяемы
            if (c == 'е') alternatives.add('ё');
            if (c == 'ё') alternatives.add('е');
            // Строим char class
            if (alternatives.size() == 1) {
                sb.append(Pattern.quote(String.valueOf(c)));
            } else {
                sb.append("[");
                for (char alt : alternatives) {
                    sb.append(Pattern.quote(String.valueOf(alt)));
                }
                sb.append("]");
            }
            // Добавляем опциональные разделители между буквами
            if (i < word.length() - 1) {
                sb.append(separators);
            }
        }
        return sb.toString();
    }
    /**
     * Фильтрует сообщение — заменяет мат на #
     */
    public FilterResult filter(String message) {
        if (!enabled || compiledPatterns.isEmpty()) {
            return new FilterResult(message, false);
        }
        String result = message;
        boolean wasFiltered = false;
        for (Pattern pattern : compiledPatterns) {
            Matcher matcher = pattern.matcher(result);
            StringBuilder sb = new StringBuilder();
            while (matcher.find()) {
                wasFiltered = true;
                String replacement = replacementChar
                    .repeat(matcher.group().length());
                matcher.appendReplacement(sb,
                    Matcher.quoteReplacement(replacement));
            }
            matcher.appendTail(sb);
            result = sb.toString();
        }
        return new FilterResult(result, wasFiltered);
    }
    public boolean isEnabled() { return enabled; }
    /**
     * Результат фильтрации
     */
    public record FilterResult(String filteredMessage, boolean wasFiltered) {}
}

package com.susworld.chat.filter;
import com.susworld.chat.SUSChatPlugin;
import org.bukkit.entity.Player;
import java.util.*;
/**
 * Система анти-спама
 * Проверяет: кулдаун, капс, повторы
 */
public class AntiSpam {
    private final SUSChatPlugin plugin;
    // UUID -> последнее время сообщения
    private final Map<UUID, Long> lastMessageTime = new HashMap<>();
    // UUID -> последнее сообщение
    private final Map<UUID, String> lastMessage = new HashMap<>();
    private boolean enabled;
    private long cooldownMs;
    private double maxCapsPercent;
    private int capsMinLength;
    private boolean blockRepeats;
    public AntiSpam(SUSChatPlugin plugin) {
        this.plugin = plugin;
        reload();
    }
    public void reload() {
        enabled = plugin.getConfig().getBoolean("antispam.enabled", true);
        cooldownMs = plugin.getConfig().getLong("antispam.cooldown", 2000);
        maxCapsPercent = plugin.getConfig().getDouble("antispam.max-caps-percent", 0.7);
        capsMinLength = plugin.getConfig().getInt("antispam.caps-min-length", 5);
        blockRepeats = plugin.getConfig().getBoolean("antispam.block-repeats", true);
    }
    /**
     * Проверяет сообщение на спам
     * @return SpamResult.OK если всё нормально
     */
    public SpamResult check(Player player, String message) {
        if (!enabled) return SpamResult.OK;
        UUID uuid = player.getUniqueId();
        // 1. Проверка кулдауна
        if (lastMessageTime.containsKey(uuid)) {
            long elapsed = System.currentTimeMillis()
                - lastMessageTime.get(uuid);
            if (elapsed < cooldownMs) {
                return SpamResult.COOLDOWN;
            }
        }
        // 2. Проверка капса
        if (message.length() >= capsMinLength) {
            long upperCount = message.chars()
                .filter(Character::isUpperCase).count();
            long letterCount = message.chars()
                .filter(Character::isLetter).count();
            if (letterCount > 0) {
                double capsRatio = (double) upperCount / letterCount;
                if (capsRatio > maxCapsPercent) {
                    return SpamResult.CAPS;
                }
            }
        }
        // 3. Проверка повторов
        if (blockRepeats && lastMessage.containsKey(uuid)) {
            if (lastMessage.get(uuid)
                    .equalsIgnoreCase(message.trim())) {
                return SpamResult.REPEAT;
            }
        }
        // Всё ок — обновляем данные
        lastMessageTime.put(uuid, System.currentTimeMillis());
        lastMessage.put(uuid, message.trim());
        return SpamResult.OK;
    }
    public enum SpamResult {
        OK, COOLDOWN, CAPS, REPEAT
    }
}

!сообщение suschat.chat.global Глобальный чат
обычное сообщение — Локальный чат (100 блоков)
/ad сообщение suschat.ad Объявление на весь сервер
/suschat reload suschat.admin Перезагрузка конфига
/suschat help suschat.admin Помощь по плагину
@ник — Упоминание игрока со звуком

🛡️ Дополнительные права

suschat.bypass.filter Обход фильтра мата
suschat.bypass.spam Обход анти-спама
suschat.ad.bypass-cooldown Обход кулдауна /ad
suschat.chat.color Использование цветов в чате

🔧 Сборка и установка

Шаги сборки

1

Установи Java 21 и Maven

JDK 21+ обязателен для Paper 1.21.1

2

Создай структуру проекта

Скопируй все файлы в соответствующие папки (см. структуру выше)

3

Собери проект

mvn clean package
4

Установи на сервер

Скопируй target/SUSChat-2.0.0.jar в папку plugins/ сервера

5

Перезапусти сервер

Плагин автоматически создаст config.yml при первом запуске

⚠️

Важно

Плагин использует Paper API и Adventure MiniMessage. Убедись, что сервер работает на Paper 1.21.1 (не Spigot/Bukkit). Adventure API включён в Paper по умолчанию.
